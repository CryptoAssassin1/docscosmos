export const CustomCard = ({ title, href, icon, children, className = "" }) => {
  const cardContent = (
    <div className={`custom-card ${className}`}>
      {icon && (
        <div className="card-icon">
          <i className={`icon-${icon}`}></i>
        </div>
      )}
      <div className="card-content">
        <h3 className="card-title">{title}</h3>
        <div className="card-description">
          {children}
        </div>
      </div>
    </div>
  );

  if (href) {
    return (
      <a href={href} className="card-link">
        {cardContent}
      </a>
    );
  }

  return cardContent;
};

export const CardGrid = ({ cols = 3, children, className = "" }) => {
  return (
    <div className={`card-grid card-grid-${cols} ${className}`}>
      {children}
    </div>
  );
};
