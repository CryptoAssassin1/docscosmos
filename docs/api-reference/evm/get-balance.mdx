<Info>
This method returns the balance of the account at the given address in wei (the smallest unit of ether).
</Info>

    ```go
    package main

    import (
        "context"
        "fmt"
        "log"
        "math/big"

        "github.com/ethereum/go-ethereum/common"
        "github.com/ethereum/go-ethereum/ethclient"
    )

    func getBalance(client *ethclient.Client, address string) (*big.Int, error) {
        account := common.HexToAddress(address)
        balance, err := client.BalanceAt(context.Background(), account, nil)
        if err != nil {
            return nil, err
        }
        return balance, nil
    }

    func main() {
        // Connect to Cosmos EVM RPC
        client, err := ethclient.Dial("https://evm-rpc.cosmos.network")
        if err != nil {
            log.Fatal(err)
        }
        defer client.Close()

        address := "0x742d35cc6600c8e8e9c94e4a9c17c2d22b14c86"
        balance, err := getBalance(client, address)
        if err != nil {
            log.Fatal(err)
        }

        fmt.Printf("Balance: %s wei\n", balance.String())

        // Convert to ETH
        fbalance := new(big.Float)
        fbalance.SetString(balance.String())
        ethValue := new(big.Float).Quo(fbalance, big.NewFloat(math.Pow10(18)))
        fmt.Printf("Balance: %s ETH\n", ethValue.String())
    }
    ```

    ```rust
    use ethers::{
        providers::{Http, Middleware, Provider},
        types::Address,
    };
    use std::convert::TryFrom;

    #[tokio::main]
    async fn main() -> Result<(), Box<dyn std::error::Error>> {
        // Connect to Cosmos EVM RPC
        let provider = Provider::<Http>::try_from("https://evm-rpc.cosmos.network")?;

        // Parse address
        let address: Address = "0x742d35cc6600c8e8e9c94e4a9c17c2d22b14c86".parse()?;

        // Get balance
        let balance = provider.get_balance(address, None).await?;

        println!("Balance: {} wei", balance);

        // Convert to ETH (18 decimals)
        let eth_balance = balance.as_u128() as f64 / 1e18;
        println!("Balance: {:.6} ETH", eth_balance);

        Ok(())
    }
    ```

    ```python
    from web3 import Web3
    import json

    # Connect to Cosmos EVM RPC
    w3 = Web3(Web3.HTTPProvider('https://evm-rpc.cosmos.network'))

    def get_balance(address: str) -> int:
        """Get balance in wei"""
        try:
            # Ensure address is checksummed
            address = w3.to_checksum_address(address)
            balance = w3.eth.get_balance(address)
            return balance
        except Exception as e:
            print(f"Error fetching balance: {e}")
            raise

    # Usage
    address = "0x742d35cc6600c8e8e9c94e4a9c17c2d22b14c86"
    balance_wei = get_balance(address)

    print(f"Balance: {balance_wei} wei")
    # Convert to ETH
    balance_eth = w3.from_wei(balance_wei, 'ether')
    print(f"Balance: {balance_eth} ETH")
    ```

## Response Format

The response returns the balance in hexadecimal format representing wei (10^-18 ETH):

```json
{
  "jsonrpc": "2.0",
  "result": "0x1b1ae4d6e2ef500000",
  "id": 1
}
```

<Note>
To convert from wei to ETH, divide the result by 10^18 (1000000000000000000).
The hex value `0x1b1ae4d6e2ef500000` equals 2000000000000000000 wei, which is 2 ETH.
</Note>

## Error Handling

Common errors and their meanings:

| Error Code | Message | Description |
|------------|---------|-------------|
| -32602 | Invalid params | The address format is invalid |
| -32000 | Unknown block | The specified block doesn't exist |
| -32001 | Account not found | The account doesn't exist on chain |

<Warning>
Always validate addresses before querying. Invalid addresses will return an error.
</Warning>
