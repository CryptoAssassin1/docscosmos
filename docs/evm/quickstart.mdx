## Smart Contract Development

### Basic Contract Example

Standard Solidity contracts should work with Cosmos EVM due to forward compatibility:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract CosmosEVMExample {
    string public message;
    address public owner;
    uint256 public updateCount;
    
    event MessageUpdated(
        string indexed newMessage, 
        address indexed updatedBy, 
        uint256 timestamp
    );
    
    constructor(string memory _message) {
        message = _message;
        owner = msg.sender;
        updateCount = 0;
        emit MessageUpdated(_message, msg.sender, block.timestamp);
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function");
        _;
    }
    
    function updateMessage(string memory _newMessage) public {
        require(bytes(_newMessage).length > 0, "Message cannot be empty");
        
        message = _newMessage;
        updateCount++;
        
        emit MessageUpdated(_newMessage, msg.sender, block.timestamp);
    }
    
    function updateMessageOwnerOnly(string memory _newMessage) public onlyOwner {
        require(bytes(_newMessage).length > 0, "Message cannot be empty");
        
        message = _newMessage;
        updateCount++;
        
        emit MessageUpdated(_newMessage, msg.sender, block.timestamp);
    }
    
    function getInfo() public view returns (
        string memory currentMessage,
        address contractOwner,
        uint256 totalUpdates,
        uint256 blockNumber,
        uint256 blockTimestamp
    ) {
        return (
            message,
            owner,
            updateCount,
            block.number,
            block.timestamp
        );
    }
}
```

### Deployment Script Template

```typescript
import { ethers } from "hardhat";

async function main() {
  console.log("Deploying to Cosmos EVM...");
  
  // Get deployer account
  const [deployer] = await ethers.getSigners();
  console.log("Deploying with account:", deployer.address);
  
  // Check balance
  const balance = await deployer.getBalance();
  console.log("Account balance:", ethers.utils.formatEther(balance), "tokens");
  
  // Deploy contract
  const CosmosEVMExample = await ethers.getContractFactory("CosmosEVMExample");
  const contract = await CosmosEVMExample.deploy("Hello Cosmos EVM!");
  
  await contract.deployed();
  
  console.log(`‚úÖ Contract deployed to: ${contract.address}`);
  console.log(`üìù Transaction hash: ${contract.deployTransaction.hash}`);
  
  // Test the contract
  const message = await contract.message();
  console.log(`üìÑ Initial message: "${message}"`);
  
  const info = await contract.getInfo();
  console.log("üìä Contract info:", {
    message: info.currentMessage,
    owner: info.contractOwner,
    updateCount: info.totalUpdates.toString(),
    blockNumber: info.blockNumber.toString(),
  });
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error("‚ùå Deployment failed:", error);
    process.exit(1);
  });
```

## Known Architecture

Based on available information, Cosmos EVM includes:

### Core Modules
- **x/vm Module**: Core EVM implementation (renamed from x/evm)
- **x/feemarket Module**: EIP-1559 fee market mechanism  
- **x/erc20 Module**: Cosmos SDK coin ‚Üî ERC-20 token interoperability

### Capabilities
- Forward-compatible with Ethereum
- Can run any valid Ethereum smart contract
- Implements features beyond standard EVM
- Integrates with Cosmos SDK for governance and IBC
