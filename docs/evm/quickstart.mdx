# EVM Quickstart Guide

This guide will help you deploy your first smart contract on Cosmos EVM in just a few minutes.

## Prerequisites

Before you begin, make sure you have:

- Node.js (v16 or higher)
- npm or yarn
- A code editor (VS Code recommended)

## Step 1: Set Up Your Environment

First, create a new directory for your project:

```bash
mkdir cosmos-evm-quickstart
cd cosmos-evm-quickstart
npm init -y
```

Install the required dependencies:

```bash
npm install --save-dev hardhat @nomicfoundation/hardhat-toolbox
npm install --save ethers dotenv
```

## Step 2: Initialize Hardhat

Create a new Hardhat project:

```bash
npx hardhat init
```

Select "Create a TypeScript project" when prompted.

## Step 3: Configure for Cosmos EVM

Update your `hardhat.config.ts` file:

```typescript
import { HardhatUserConfig } from "hardhat/config";
import "@nomicfoundation/hardhat-toolbox";
import * as dotenv from "dotenv";

dotenv.config();

const config: HardhatUserConfig = {
  solidity: {
    version: "0.8.19",
    settings: {
      optimizer: {
        enabled: true,
        runs: 200,
      },
    },
  },
  networks: {
    "cosmos-testnet": {
      url: "https://rpc-evm-testnet.cosmos.network",
      accounts: process.env.PRIVATE_KEY ? [process.env.PRIVATE_KEY] : [],
      chainId: 9000,
      gas: 2100000,
      gasPrice: 8000000000,
    },
  },
};

export default config;
```

Create a `.env` file in your project root:

```bash
PRIVATE_KEY=your_private_key_here
```

<Warning>
Never commit your private key to version control. Add `.env` to your `.gitignore` file.
</Warning>

## Step 4: Write Your Smart Contract

Create a simple contract in `contracts/HelloCosmos.sol`:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.19;

contract HelloCosmos {
    string public message;
    address public owner;
    uint256 public updateCount;
    
    event MessageUpdated(string newMessage, address updatedBy, uint256 timestamp);
    
    constructor(string memory _initialMessage) {
        message = _initialMessage;
        owner = msg.sender;
        updateCount = 0;
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Only owner can call this function");
        _;
    }
    
    function updateMessage(string memory _newMessage) public {
        message = _newMessage;
        updateCount++;
        emit MessageUpdated(_newMessage, msg.sender, block.timestamp);
    }
    
    function updateMessageOwnerOnly(string memory _newMessage) public onlyOwner {
        message = _newMessage;
        updateCount++;
        emit MessageUpdated(_newMessage, msg.sender, block.timestamp);
    }
    
    function getMessage() public view returns (string memory) {
        return message;
    }
    
    function getContractInfo() public view returns (
        string memory currentMessage,
        address contractOwner,
        uint256 totalUpdates,
        uint256 blockTimestamp
    ) {
        return (message, owner, updateCount, block.timestamp);
    }
}
```

## Step 5: Write Deployment Script

Create `scripts/deploy.ts`:

```typescript
import { ethers } from "hardhat";

async function main() {
  const HelloCosmos = await ethers.getContractFactory("HelloCosmos");
  
  console.log("Deploying HelloCosmos contract...");
  
  const helloCosmos = await HelloCosmos.deploy("Hello from Cosmos EVM!");
  
  await helloCosmos.deployed();
  
  console.log(`HelloCosmos deployed to: ${helloCosmos.address}`);
  console.log(`Transaction hash: ${helloCosmos.deployTransaction.hash}`);
  
  // Wait for a few block confirmations
  await helloCosmos.deployTransaction.wait(3);
  
  console.log("Contract deployed and confirmed!");
  
  // Test the contract
  const message = await helloCosmos.getMessage();
  console.log(`Initial message: ${message}`);
  
  const info = await helloCosmos.getContractInfo();
  console.log(`Contract info:`, {
    message: info.currentMessage,
    owner: info.contractOwner,
    updateCount: info.totalUpdates.toString(),
    timestamp: new Date(info.blockTimestamp.toNumber() * 1000).toISOString()
  });
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
```

## Step 6: Deploy Your Contract

Compile your contract:

```bash
npx hardhat compile
```

Deploy to Cosmos EVM testnet:

```bash
npx hardhat run scripts/deploy.ts --network cosmos-testnet
```

You should see output similar to:


## Step 7: Interact with Your Contract

Create `scripts/interact.ts`:

```typescript
import { ethers } from "hardhat";

async function main() {
  const contractAddress = "0x1234567890123456789012345678901234567890"; // Replace with your deployed address
  
  const HelloCosmos = await ethers.getContractFactory("HelloCosmos");
  const helloCosmos = HelloCosmos.attach(contractAddress);
  
  // Read current message
  console.log("Current message:", await helloCosmos.getMessage());
  
  // Update message
  console.log("Updating message...");
  const tx = await helloCosmos.updateMessage("Updated from TypeScript!");
  await tx.wait();
  
  console.log("New message:", await helloCosmos.getMessage());
  
  // Get contract info
  const info = await helloCosmos.getContractInfo();
  console.log("Update count:", info.totalUpdates.toString());
}

main()
  .then(() => process.exit(0))
  .catch((error) => {
    console.error(error);
    process.exit(1);
  });
```

Run the interaction script:

```bash
npx hardhat run scripts/interact.ts --network cosmos-testnet
```
