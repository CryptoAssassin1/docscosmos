# EVM Development on Cosmos

Deploy Solidity contracts with full EVM compatibility while benefiting from Cosmos' parallel transaction processing and rich interoperability.

## Quick Start

Get started with EVM development in just a few steps:

### 1. Install Dependencies

```bash
npm install --save ethers
npm install --save @cosmos/cosmjs
```

### 2. Deploy a Smart Contract

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract HelloCosmos {
    string public message;
    
    constructor(string memory _message) {
        message = _message;
    }
    
    function updateMessage(string memory _newMessage) public {
        message = _newMessage;
    }
    
    function getMessage() public view returns (string memory) {
        return message;
    }
}
```

### 3. Interact with Your Contract

```typescript
import { ethers } from 'ethers';

// Connect to the Cosmos EVM chain
const provider = new ethers.providers.JsonRpcProvider('https://rpc-evm.cosmos.network');
const wallet = new ethers.Wallet(process.env.PRIVATE_KEY!, provider);

// Contract deployment
async function deployContract() {
    const HelloCosmos = await ethers.getContractFactory("HelloCosmos", wallet);
    const contract = await HelloCosmos.deploy("Hello from Cosmos!");
    
    await contract.deployed();
    console.log("Contract deployed to:", contract.address);
    
    return contract;
}

// Interact with the contract
async function interactWithContract(contractAddress: string) {
    const contract = new ethers.Contract(
        contractAddress,
        ["function getMessage() view returns (string)", "function updateMessage(string)"],
        wallet
    );
    
    // Read the current message
    const currentMessage = await contract.getMessage();
    console.log("Current message:", currentMessage);
    
    // Update the message
    const tx = await contract.updateMessage("Updated from TypeScript!");
    await tx.wait();
    
    console.log("Message updated successfully!");
}
```
